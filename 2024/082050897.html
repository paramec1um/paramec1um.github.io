1 二维几何基本运算

1.1 定义点与向量

#define IN inline
struct Node{
    double x,y;

    //排序为Andrew凸包算法做准备
    IN bool operator<(const Node &t)const {//按照x排序
        if(x!=t.x) return x<t.x;
        return y<t.y;//x相同按照y排序
    }

    IN Node operator+(const Node &t)const {return {x+t.x,y+t.y};}
    IN Node operator-(const Node &t)const {return {x-t.x,y-t.y};} 
    IN Node operator*(const double &t)const {return {x*t,y*t};}//数乘
    IN double operator*(const Node &t)const {return x*t.y-y*t.x;} //叉积 
    IN double operator^(const Node &t)const {return x*t.x+y*t.y;} //点积 
};
IN double dis(Node a){//距离
    return sqrt(a.x*a.x+a.y*a.y);
}

1.2 定义点向式直线

struct Line{
    Node u,v;//u为直线上一点，v为方向向量，在模板里面为逆时针指向，也即在x轴正方向的投影与x轴正方向同向
	double a=atan2(v.y,v.x);
	IN bool operator<(const Line t)const {//按照极角排序
		if(a==t.a)return (t.u-u)*v>0;
        //若构建下半面的半平面，极角相同，从左到右排序，改为>0
        //否则若构造上半面的半平面，极角相同，从右到左排序，改为<0
		return a<t.a;
	}
	IN bool operator !=(const Line t)const {return a!=t.a;}
};
IN Node Rotation(Node v){//逆时针旋转90度
    return {-v.y,v.x};
}
IN Node inter(Line l1,Line l2){//求交点
    //此处没有避免平行和0向量的情况，需要在代码solve部分解决
    double t=(l2.v*(l2.u-l1.u))/(l2.v*l1.v);
    return l1.u+l1.v*t;
}

1.3 计算n边形的面积

已知 $n$ 边形所有顶点的坐标
用 Node d[n] 表示这些点，则可以通过高斯面积公式计算其面积为：

IN double S(Node d[])
    double s=0;
    for(int i=1;i<n;++i){
        s+=d[i]*d[i+1];//注意内部的加法是不带绝对值的
    }
    s+=d[n]*d[1];
    return fabs(s)/2;//确保面积为正数

2 凸包相关

2.1 凸包构造

Andrew算法：
先按 $x,y$ 的优先顺序排序
之后从前往后扫一次，求下半凸包
再从后往前扫一次，求上半凸包

当前点在直线下方时弹栈，否则加入

而 $a\times b =|a|*|b|*sin(\theta)$ 其中 $\theta$ 表示 向量 $a$ 逆时针旋转到达 $b$ 时转过的角度，当这个角度是负数时代表点在直线下方 

int top,n;//n记录点数，top记录栈大小，栈尾有重复的1号点
node a[maxn],stk[maxn];
void Andrew(){
    sort(a+1,a+1+n);
    for(int i=1;i<=n;++i){
        while(top>1 && ((a[i]-stk[top-1])*(stk[top]-stk[top-1]))<=0)--top;
        stk[++top]=a[i];
    }
    int tmp=top;
    for(int i=n;i>=1;--i){
        while(top>tmp && ((a[i]-stk[top-1])*(stk[top]-stk[top-1]))<=0)--top;
        stk[++top]=a[i];
    }
    --top;//去掉重复的1号点
}

2.2 旋转卡壳

2.2.1 求二维凸包直径

一般来说，要先用Andrew算法求凸包

先利用叉积大小确定离初始边最远的的点
之后开始旋转，不断更新答案

double get_long(){//旋转卡壳求直径
    double ans=0;//去掉重复的1号点
    for(int i=1,j=2;i<=top;++i){
        while(fabs((stk[i]-stk[i+1])*(stk[i]-stk[j]))<fabs((stk[i]-stk[i+1])*(stk[i]-stk[j%top+1])))
            j=j%top+1;//寻找最远点
        ans=max(ans,max(dis(st[i]-st[j]),dis(st[i+1]-st[j])));
    }
    return ans;
}

2.2.2 求最小矩形覆盖

  1. 利用旋转卡壳求出最高点
  2. 利用往右的点一定是点积由正到负，往左的点的点积一定由负到正求出左右点
  3. 利用直线相交求出交点与面积，或者直接求面积 
    double ans=1e18;
void cal(int i,int l,int r,int j){//面积,求交点参考1.0.2
    double t1=fabs((stk[i]-stk[i+1])*(stk[i]-stk[j]));
    double t2=fabs((stk[i]-stk[i+1])^(stk[i]-stk[r]))+fabs((stk[i]-stk[i+1])^(stk[i]-stk[l]));
    double t3=fabs((stk[i]-stk[i+1])^(stk[i]-stk[i+1]));
    ans=min(ans,t1*t2/t3);
}
void get_s(){
    for(int i=1,j=2,r=2,l=2;i<=top;++i){
        while(fabs((stk[i]-stk[i+1])*(stk[i]-stk[j]))<fabs((stk[i]-stk[i+1])*(stk[i]-stk[j%top+1])))j=j%top+1;
        l=max(l,j);
        while(((stk[i+1]-stk[i])^(stk[r%top+1]-stk[r]))>=0)r=r%top+1;
        while(((stk[i+1]-stk[i])^(stk[l%top+1]-stk[l]))<=0)l=l%top+1;
        cal(i,l,r,j);//计算面积或者交点
    }
}

2.2.3求最大内接四边形

旋转卡壳可以求凸包内最大内街四边形

通过对角线确定最大内接四边形
在维护对角线的时候，利用最大内接四边形的特点，用类似旋转卡壳的思想维护另一条对角线 $O(nlogn)$

double ans=0;
void get_s_in(){
    for(int u=1,v=2,r=1,l=2;u<=top;++u){
        while((stk[u%top+1]-stk[u])*(stk[v%top+1]-stk[v])<0){//找对角线
            if(l==v)l=l%top+1;//保证l在v的后面
            v=v%top+1;
        } 
        while(l%top+1!=u&&fabs((stk[u]-stk[v])*(stk[l]-stk[v]))<fabs((stk[u]-stk[v])*(stk[l%top+1]-stk[v])))
            l=l%top+1;
        while(r%top+1!=v&&fabs((stk[u]-stk[v])*(stk[r]-stk[v]))<fabs((stk[u]-stk[v])*(stk[r%top+1]-stk[v])))
            r=r%top+1;
        ans=max(ans,fabs((stk[u]-stk[v])*(stk[r]-stk[v]))+fabs((stk[u]-stk[v])*(stk[l]-stk[v])));
        if(r==u)r=u%top+1;
    }
}

3 半平面交

3.1 半平面交求面积

在点向直线的基础上，选择构造下半平面
先对极角进行排序
在保证极角不同的情况下，对于当前预插入向量：

  1. 先弹出队尾，若上一个交点在向量右侧则弹出
  2. 弹出队首，若第一个交点在向量右侧则弹出

之后插入当前向量，求出新的交点

所有向量插入完毕后，删除多余的元素

//输出半平面交的面积
//e[]存输入的向量
//q[]为单调队列
//(l,r]维护单调队列的有效范围，左开右闭
//ans[]存合法点集
Line e[maxn];
solve(){
    sort(e+1,e+1+tot);//排序
	int l = 0, r = 0;
	for (int i = 1; i <= tot; ++i)
  		if (e[i] != e[i - 1]) {
    		// 注意要先检查队尾
    		while (r - l > 1 && e[i].v * (e[i].u - ans[r]) >0)  // 如果上一个交点在向量右侧则弹出队尾
      			--r;
    		while (r - l > 1 && e[i].v * (e[i].u - ans[l + 2]) >0)  // 如果第一个交点在向量右侧则弹出队首
      			++l;
    		q[++r] = e[i];
    		if (r - l > 1) ans[r] = inter(q[r], q[r - 1]);  // 求新交点
  		}
	while (r - l > 1 && q[l+1].v * (q[l + 1].u - ans[r]) > 0)  // 注意删除多余元素
  		--r;
	if(r<=l+2)cout<<"0.000\n";//剩余的点只有1个或者没有的情况，直接输出0
	else{
	ans[r + 1] = inter(q[l + 1], q[r]);  // 再求出新的交点
	++r;
	double sum=0;
	for(int i=l+3;i<=r;++i)sum+=ans[i-1]*ans[i];//逆时针三角剖分求出面积
	sum+=ans[r]*ans[l+2];
	cout<<fixed<<setprecision(3);
	cout<<fabs(sum/2);
	}
}

3.2 半平面交确定可见直线

从上方往下看，作上半面的半平面交
从下往上看，作下半面半平面交
之后按照极角排序
对于一个预插入的直线

  • 若栈中元素小于两个直接插入
  • 若当前直线与栈顶top直线的交点在上上一个top-1直线的下方则弹出栈顶元素
  • 否则插入当前直线
    //从大到小输出可见直线的序号
//e[]记录输入直线
//stk[]维护动态栈
//ans[]记录最终答案
int stk[maxn]
void solve(){
    sort(e+1,e+1+n);
    int top=0;
    for(int i=1;i<=n;++i){
        if(i==n||e[i]!=e[i+1]){//排除相同的直线
            //此处的inter写内部，防止top为0时出现与0向量相交的情况
            while(top>1&&(inter(e[i],e[stk[top]])-e[stk[top-1]].u)*e[stk[top-1]].v>=0)top--;
            stk[++top]=i;
            ans[top]=e[i].id;
        }
    }
    for(int i=1;i<=top;++i)ans[i]=e[stk[i]].id;
    sort(ans+1,ans+1+top);
    for(int i=1;i<=top;++i)
        cout<<ans[i]<<' ';
}
